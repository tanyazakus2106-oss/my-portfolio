**Frontmatter (exact):**
```ts
---
import BaseLayout from '../../layouts/BaseLayout.astro';
import FullBleedImage from '../../components/FullBleedImage.astro';
import { Image } from 'astro:assets';
import { getCollection, render } from 'astro:content';
import type { CollectionEntry } from 'astro:content';

export async function getStaticPaths() {
  const all = (await getCollection('projects'))
    .sort((a, b) => b.data.publishDate.valueOf() - a.data.publishDate.valueOf());

  return all.map((entry, i) => ({
    params: { id: entry.id },
    props: {
      entry,
      // Wrap-around: "Next project →" is NEWER (lower index, wraps to oldest).
      // "← Previous project" is OLDER (higher index, wraps to newest).
      nextEntry: all[(i - 1 + all.length) % all.length],
      prevEntry: all[(i + 1) % all.length],
    },
  }));
}

type Props = {
  entry: CollectionEntry<'projects'>;
  nextEntry: CollectionEntry<'projects'>;
  prevEntry: CollectionEntry<'projects'>;
};

const { entry, nextEntry, prevEntry } = Astro.props as Props;
const { Content } = await render(entry);

// Timeline formatting — placeholder MDX has only publishDate, so we render the month+year
// as a single-point timeline. Real case studies can override by adding explicit timeline
// text in MDX body if needed.
const timeline = entry.data.publishDate.toLocaleDateString('en-US', {
  month: 'long',
  year: 'numeric',
});
---
```

**IMPORTANT — animating section blocks (D-20):**
MDX h2 headings render as bare `<h2>` inside the `<Content />` output. To wrap each "section block" (h2 + following paragraphs until next h2) in `.animate-on-scroll` without post-processing, the simplest correct approach is:

**Apply `.animate-on-scroll` to each `h2` AND require the MDX author to wrap each section's body in a `<section class="animate-on-scroll">` block.**

Instead of that complexity, for the PLACEHOLDER phase the simpler contract is:
- Add `.animate-on-scroll` as a class to each `<h2>` via a global CSS rule scoped to `.case-prose h2`.
- The paragraphs beneath will be visible from the start; only the section heading fades in.
- This satisfies D-20 "case study section blocks animate in" at the heading-level granularity (the section itself is marked by its heading).

Add to the `<style is:global>` block above, below `.case-prose p`:
```css
.case-prose h2 {
  /* Animate-on-scroll participation — D-20. */
  opacity: 0;
  transform: translateY(24px);
  transition:
    opacity 500ms ease-out,
    transform 500ms ease-out;
  transition-delay: calc(var(--stagger-index, 0) * 60ms);
}
.case-prose h2.is-visible {
  opacity: 1;
  transform: translateY(0);
}
@media (prefers-reduced-motion: reduce) {
  .case-prose h2 {
    opacity: 1;
    transform: none;
    transition: none;
  }
}
```

**And update `scroll-animation.ts` contract:** The existing script selects `.animate-on-scroll`. Since we DON'T put the raw class on the h2 (to avoid MDX authoring friction), we need the script to ALSO observe `.case-prose h2`.

**Simplest correct solution:** Add a small inline script at the bottom of `[id].astro` (module mode) that imports the shared scroll-animation AND additionally observes `.case-prose h2`. Replace the `<script src="...">` with:

```astro
<script>
  // Observe MDX section headings on case study pages — extends the shared
  // scroll-animation behavior. D-20, POL-01.
  import '../../scripts/scroll-animation.ts';

  const h2s = document.querySelectorAll<HTMLElement>('.case-prose h2');
  if (h2s.length > 0) {
    const prefersReducedMotion =
      window.matchMedia('(prefers-reduced-motion: reduce)').matches;
    if (prefersReducedMotion) {
      h2s.forEach((el) => el.classList.add('is-visible'));
    } else {
      h2s.forEach((el, i) => {
        el.style.setProperty('--stagger-index', String(i));
      });
      const observer = new IntersectionObserver(
        (entries) => {
          entries.forEach((entry) => {
            if (entry.isIntersecting) {
              entry.target.classList.add('is-visible');
              observer.unobserve(entry.target);
            }
          });
        },
        { threshold: 0.1, rootMargin: '0px 0px -40px 0px' }
      );
      h2s.forEach((el) => observer.observe(el));
    }
  }
</script>
```
