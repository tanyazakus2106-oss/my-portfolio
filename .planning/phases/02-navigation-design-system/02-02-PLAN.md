```astro
<script>
  const trigger = document.getElementById('mobile-nav-trigger') as HTMLButtonElement | null;
  const overlay = document.getElementById('mobile-nav-overlay') as HTMLDivElement | null;
  const closeBtn = document.getElementById('mobile-nav-close') as HTMLButtonElement | null;

  if (trigger && overlay && closeBtn) {
    let isOpen = false;

    function getFocusable(): HTMLElement[] {
      if (!overlay) return [];
      return Array.from(
        overlay.querySelectorAll<HTMLElement>(
          'a[href], button:not([disabled])'
        )
      );
    }

    function open() {
      if (!overlay || !trigger) return;
      isOpen = true;
      overlay.dataset.state = 'open';
      overlay.classList.remove('opacity-0', 'pointer-events-none');
      overlay.classList.add('opacity-100');
      trigger.setAttribute('aria-expanded', 'true');
      trigger.setAttribute('aria-label', 'Close navigation');
      document.body.style.overflow = 'hidden';
      // Focus first focusable (close button) on next tick so transition has started
      requestAnimationFrame(() => {
        const focusables = getFocusable();
        focusables[0]?.focus();
      });
    }

    function close() {
      if (!overlay || !trigger) return;
      isOpen = false;
      overlay.dataset.state = 'closed';
      overlay.classList.add('opacity-0', 'pointer-events-none');
      overlay.classList.remove('opacity-100');
      trigger.setAttribute('aria-expanded', 'false');
      trigger.setAttribute('aria-label', 'Open navigation');
      document.body.style.overflow = '';
      trigger.focus();
    }

    trigger.addEventListener('click', open);
    closeBtn.addEventListener('click', close);

    // Backdrop click: close if click target is the overlay itself (not a child)
    overlay.addEventListener('click', (e) => {
      if (e.target === overlay) close();
    });

    // Link click closes overlay so SPA-ish feel works (and resets for next page)
    overlay.querySelectorAll<HTMLAnchorElement>('[data-mobile-nav-link]').forEach((a) => {
      a.addEventListener('click', close);
    });

    // Esc + focus trap
    document.addEventListener('keydown', (e) => {
      if (!isOpen) return;
      if (e.key === 'Escape') {
        e.preventDefault();
        close();
        return;
      }
      if (e.key === 'Tab') {
        const focusables = getFocusable();
        if (focusables.length === 0) return;
        const first = focusables[0];
        const last = focusables[focusables.length - 1];
        const active = document.activeElement as HTMLElement | null;
        if (e.shiftKey && active === first) {
          e.preventDefault();
          last.focus();
        } else if (!e.shiftKey && active === last) {
          e.preventDefault();
          first.focus();
        }
      }
    });
  }
</script>
```
