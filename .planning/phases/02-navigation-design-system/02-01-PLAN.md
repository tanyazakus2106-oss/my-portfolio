```astro
  <svg class="theme-icon-sun h-5 w-5 hidden" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" aria-hidden="true"><circle cx="12" cy="12" r="4"/><path d="M12 2v2"/><path d="M12 20v2"/><path d="m4.93 4.93 1.41 1.41"/><path d="m17.66 17.66 1.41 1.41"/><path d="M2 12h2"/><path d="M20 12h2"/><path d="m4.93 19.07 1.41-1.41"/><path d="m17.66 6.34 1.41-1.41"/></svg>
</button>

<style>
  :global(html.dark) .theme-icon-moon { display: none; }
  :global(html.dark) .theme-icon-sun { display: inline-block; }
</style>

<script>
  const btn = document.getElementById('theme-toggle');
  const root = document.documentElement;

  function syncAria() {
    const isDark = root.classList.contains('dark');
    btn?.setAttribute('aria-label', isDark ? 'Switch to light mode' : 'Switch to dark mode');
  }

  syncAria();

  btn?.addEventListener('click', () => {
    const willBeDark = !root.classList.contains('dark');
    root.classList.toggle('dark', willBeDark);
    localStorage.setItem('theme', willBeDark ? 'dark' : 'light');
    syncAria();
  });
</script>
```

Step B — Rewrite `src/layouts/BaseLayout.astro` so that:

1. It imports Header and Footer components
2. It includes an inline FOUC-prevention `<script is:inline>` as the FIRST child of `<head>` — BEFORE the Google Fonts preconnect/link tags and BEFORE any CSS

Full rewritten file:
```astro
 ---
import "../styles/global.css";
import Header from "../components/Header.astro";
import Footer from "../components/Footer.astro";

interface Props {
  title?: string;
  description?: string;
}

const {
  title = "Tanya Zakus — UX/UI Designer",
  description = "",
} = Astro.props;
 ---

<!DOCTYPE html>
<html lang="en">
  <head>
    <script is:inline>
      // FOUC prevention — D-02. Must run synchronously before any CSS or body renders.
      (function () {
        try {
          var stored = localStorage.getItem('theme');
          var prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
          var isDark = stored ? stored === 'dark' : prefersDark;
          if (isDark) document.documentElement.classList.add('dark');
        } catch (e) {
          // localStorage unavailable — fall back to system preference via CSS @media
        }
      })();
    </script>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <title>{title}</title>
    {description && <meta name="description" content={description} />}
    <link rel="preconnect" href="https://fonts.googleapis.com" />
    <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
    <link
      href="https://fonts.googleapis.com/css2?family=DM+Sans:wght@400;600&display=swap"
      rel="stylesheet"
    />
  </head>
  <body class="font-sans text-[var(--color-text-primary)] bg-[var(--color-dominant)]">
    <Header />
    <main>
      <slot />
    </main>
    <Footer />
  </body>
</html>
```

Critical: the `<script is:inline>` MUST be the FIRST child of `<head>` (before `<meta charset>` is acceptable per spec, but the key constraint is BEFORE the CSS `<link>` and BEFORE body renders). `is:inline` tells Astro to NOT bundle/defer this script — it inlines verbatim and runs synchronously at parse time.
